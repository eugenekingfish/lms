\newtcolorbox{coolbox}[1]{
   colframe=black!75!white, 
   colback=black!10!white,
   boxrule=1pt,
   halign=flush left,
   breakable,
   sharp corners, 
   title = {\centering\textbf{\textrm{#1}}},
   arc = 0.001mm, % fixes display problems with some pdf viewers
   subtitle style = {
      boxrule=1pt,
      colback=black!50!white,
   },
   segmentation style={purple!75!white, solid, line width=0.5pt, opacity=0.5}
}
